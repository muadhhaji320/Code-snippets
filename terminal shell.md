

```shell
# Change directory to a specified path
cd /path/to/directory

# List files and directories in the current directory
ls

# Create a new directory
mkdir directory_name 

# Copy a file from one location to another
cp source_file destination_file

# Move or rename a file or directory
mv old_name new_name

# Remove a file
rm file_name

# Remove a directory and its contents recursively
rm -r directory_name

# Print the current working directory
pwd

# Display the contents of a file
cat file_name

# Display the first few lines of a file
head file_name

# Display the last few lines of a file
tail file_name

# Search for a specific pattern in a file
grep pattern file_name

# Display the manual page for a command
man command_name

# Execute a command with superuser privileges
sudo command

# Download a file from the internet
wget URL

# Display the IP address of the current machine
ipconfig (Windows)
ifconfig (Linux/Mac)

# Display network information
netstat

# Connect to a remote server using SSH
ssh username@remote_host

# Archive files into a compressed format
tar -cvf archive_name.tar files_to_archive

# Extract files from a compressed archive
tar -xvf archive_name.tar

# Display the current date and time
date

# Display system information
uname -a
# Display the system's uptime, including current time,
# how long the system has been running,
# and the number of users logged in.
uptime
# display the username of the current user who is logged into the system.
$ whoami
john
# displays information about currently logged-in users,
# including their username, terminal, login time, and more
$ who
john     tty1         2024-05-31 10:00 (:0)
```
